---
title: "Create Plan"
api: "POST /plan/"
description: "Create a new meal plan for the authenticated user"
---

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

```
Authorization: Bearer <token>
```

## Request Body

<ParamField body="daily_menus" type="array" default="[]">
  List of daily menus to include in the plan
  
  <ParamField body="day_of_week" type="integer" required>
    Day of the week (1-7, where 1 is Monday)
  </ParamField>
  
  <ParamField body="plan_id" type="integer" required>
    Plan ID to which this daily menu belongs
  </ParamField>
  
  <ParamField body="meal_details" type="array" default="[]">
    List of meal details for this day
    
    <ParamField body="recipe_id" type="integer" required>
      Recipe ID associated with the meal detail
    </ParamField>
    
    <ParamField body="schedule" type="integer" required>
      Order of meals (1-6)
    </ParamField>
    
    <ParamField body="status" type="integer" required>
      Status of the meal detail (0: pending, 1: completed)
    </ParamField>
    
    <ParamField body="meal_type" type="string" required>
      Type of meal. Possible values: `breakfast`, `lunch`, `dinner`, `snack`
    </ParamField>
  </ParamField>
</ParamField>

## Error Responses

<ResponseField name="401 Unauthorized">
  The request lacks valid authentication credentials
  
  ```json
  {
    "detail": "Not authenticated"
  }
  ```
</ResponseField>

<ResponseField name="422 Unprocessable Entity">
  Invalid request body or validation error
  
  ```json
  {
    "detail": [
      {
        "loc": ["body", "daily_menus", 0, "day_of_week"],
        "msg": "ensure this value is greater than or equal to 1",
        "type": "value_error.number.not_ge"
      }
    ]
  }
  ```
</ResponseField>

<ResponseField name="500 Internal Server Error">
  An unexpected error occurred on the server
  
  ```json
  {
    "detail": "Internal server error"
  }
  ```
</ResponseField>
